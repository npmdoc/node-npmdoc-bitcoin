function bitcoin.Client (opts)

    function Client(opts) {
  this.rpc = new rpc.Client(opts)

## Examples

### Create client
```js
// all config options are optional
var client = new bitcoin.Client({
  host: 'localhost',
  port: 8332,
  user: 'username',
  pass: 'password',
  timeout: 30000
});
```
